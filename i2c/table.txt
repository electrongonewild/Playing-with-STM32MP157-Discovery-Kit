CREATE TABLE LocalAuthorizationList (
            ID_Token_Value  	   CHAR(50) PRIMARY KEY NOT NULL,
            ID_Token_Type          CHAR(20)             NOT NULL,
            Status                 CHAR(20)                     		
        );

CREATE TABLE Master (
	    Parameter	           CHAR(50)             NOT NULL,
            Value	  	   CHAR(50) 		NOT NULL		
        );

CREATE TABLE Cache (
            ID_Token_Value         CHAR(50) PRIMARY KEY NOT NULL,
            ID_Token_Type          CHAR(20)             NOT NULL,
            Status                 CHAR(20)                     ,
            Cache_Expiry_Date_Time CHAR(30)                     ,
            Timestamp              CHAR(30)             NOT NULL
        );
